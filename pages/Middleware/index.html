  • @Middleware -- 修饰类为中间件,类的成员方法都将被 koa ues 加载
  • @Order -- 中间件加载顺序(级别)
@Middleware
class IndexMiddleware {
    @Order(1)
    async index(ctx, next) {
         console.log("我会第一个被加载")
         await next()
    }
     @Order(2)
    async index2(ctx, next) {
         console.log("我会第二个被加载")
         await next()
    }
}
